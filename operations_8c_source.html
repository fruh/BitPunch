685 int BPU_gf2xPolyIrredTest(const BPU_T_Poly_GF2_16x *p, const BPU_T_Arithmetic_Data *a_data) {
686  // x^(q^n) = x^((2^m)^n) = x^(2^(m*n))
687  // q = 2^m
688  // check if gcd(x^(q^n) - x, f)==f. if not, f is reducible
689  // find all prime factors of n = deg(f)
690  // for every prime factor t of n check if gcd(x^(q^(n/t)), f) == 1. if not f is reducible
691  // otherwise f is ireducible
692  int i, j;
693  int is_irred = 1;
694  int m = a_data->mod_deg;
695  int n = p->deg;
696  int exponent = m*n;
697  BPU_T_Poly_GF2_16x tmp, out, qr, x, gcd, s, t, one;
698 
699  // test if some alpha is root
700  for (i = 0; i < a_data->ord; i++) {
701  if (BPU_gf2xPolyEval(p, a_data->exp_table[i], a_data) == 0) {
702  is_irred = 0;
703  return is_irred;
704  }
705  }
706  BPU_mallocPoly(&out, 0);
707  BPU_mallocPoly(&one, 0);
708  BPU_mallocPoly(&qr, 0);
709  BPU_mallocPoly(&tmp, 1);
710  BPU_mallocPoly(&x, 1);
711  // gcd, s, t, will be allocated inside gf2xPolyExtEuclidA
712 
713  // set tmp polynomial tmp(x) = x
714  tmp.coef[0] = 0;
715  tmp.coef[1] = 1;
716  tmp.deg = BPU_gf2xPolyGetDeg(&tmp);
717 
718  x.coef[0] = 0;
719  x.coef[1] = 1;
720  x.deg = BPU_gf2xPolyGetDeg(&x);
721 
722  one.coef[0] = 1;
723  one.deg = 0;
724 
725  // simplify polynomial with big coeffitient
726  for (i = 0; i < exponent; i++) {
727  BPU_gf2xPolyMul(&out, &tmp, &tmp, a_data);
728  BPU_gf2xPolyMod(&tmp, &out, p, a_data);
729  }
730  BPU_gf2xPolyAdd(&qr, &tmp, &x);
731  BPU_gf2xPolyExtEuclidA(&gcd, &s, &t, &qr, p, -1, a_data);
732 
733  if (BPU_gf2xPolyCmp(p, &gcd) != 1) {
734  BPU_freePoly(&out, 0);
735  BPU_freePoly(&one, 0);
736  BPU_freePoly(&qr, 0);
737  BPU_freePoly(&tmp, 0);
738  BPU_freePoly(&x, 0);
739  BPU_freePoly(&gcd, 0);
740  BPU_freePoly(&s, 0);
741  BPU_freePoly(&t, 0);
742 
743  is_irred = 0;
744  return is_irred;
745  }
746 
747  for (j = 2; j <= p->deg; j++) {
748  if (p->deg % j != 0 || BPU_isPrime(j) == 0)
749  continue;
750 
751  BPU_gf2xPolyNull(&tmp);
752  tmp.coef[0] = 0;
753  tmp.coef[1] = 1;
754  tmp.deg = BPU_gf2xPolyGetDeg(&tmp);
755 
756  exponent = m*(n/j);
757 
758  for (i = 0; i < exponent; i++) {
759  BPU_gf2xPolyMul(&out, &tmp, &tmp, a_data);
760  BPU_gf2xPolyMod(&tmp, &out, p, a_data);
761  }
762  BPU_freePoly(&gcd, 0);
763  BPU_freePoly(&s, 0);
764  BPU_freePoly(&t, 0);
765 
766  BPU_gf2xPolyExtEuclidA(&gcd, &s, &t, &tmp, p, -1, a_data);
767 
768  if (BPU_gf2xPolyCmp(&one, &gcd) != 1) {
769  is_irred = 0;
770  break;
771  }
772  }
773  BPU_freePoly(&out, 0);
774  BPU_freePoly(&one, 0);
775  BPU_freePoly(&qr, 0);
776  BPU_freePoly(&tmp, 0);
777  BPU_freePoly(&x, 0);
778  BPU_freePoly(&gcd, 0);
779  BPU_freePoly(&s, 0);
780  BPU_freePoly(&t, 0);
781 
782  return is_irred;
